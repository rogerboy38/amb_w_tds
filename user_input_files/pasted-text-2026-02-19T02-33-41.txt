TASK 4: Update _extract_variant() in parser.py
Modify _extract_variant() to ALSO check SLASH_RATIO_PATTERNS for Highpol/Acetypol families.

Logic:

If family is HIGHPOL or ACETYPOL → check SLASH_RATIO_PATTERNS first, then RATIO_PATTERNS

If family is 0227 or 0307 → check RATIO_PATTERNS only (existing behavior)

Validate extracted variant against valid_variants list

Fall back to default_concentration

python
def _extract_variant(self, request_text: str, family: str) -> Optional[str]:
    family_info = self.PRODUCT_FAMILIES.get(family, {})
    valid_variants = family_info.get("valid_variants", [])
    default = family_info.get("default_concentration")

    # For Highpol/Acetypol, try slash patterns first
    if family in ("HIGHPOL", "ACETYPOL"):
        for pattern, normalizer in self.SLASH_RATIO_PATTERNS:
            match = pattern.search(request_text)
            if match:
                normalized = normalizer(match)
                if valid_variants and normalized in valid_variants:
                    return normalized
    
    # Try standard ratio patterns (30:1 -> 30X)
    for pattern, normalizer in self.RATIO_PATTERNS:
        match = pattern.search(request_text)
        if match:
            normalized = normalizer(match)
            if valid_variants:
                if normalized in valid_variants:
                    return normalized
                for vv in valid_variants:
                    if normalized.upper() == vv.upper():
                        return vv
            else:
                return normalized

    # Return default if family has variants
    if family_info.get("has_variants") and default:
        return default

    return None
